Seleziona gli utenti che hanno postato almeno un video

select distinctrow  u.id , u.username 
from users u 
inner join  medias m  on u.id = m.user_id
WHERE  m.`type` = 'video';


Seleziona tutti i post senza Like

SELECT p.id, p.title
FROM posts p
LEFT JOIN likes l ON p.id = l.post_id
WHERE l.post_id IS NULL;



Conta il numero di like per ogni post (165 nel caso di query con SELECT + JOIN oppure 175 nel caso di query con sola SELECT)


SELECT p.id, p.title, COUNT(l.post_id) AS like_count
FROM posts p
LEFT JOIN likes l ON p.id = l.post_id
GROUP BY p.id, p.title;


SELECT post_id, COUNT(*) AS like_count
FROM likes
GROUP BY post_id;


Ordina gli utenti per il numero di media caricati (25) 

SELECT u.username , COUNT( m.id) AS media_count
FROM users u 
LEFT JOIN medias m ON u.id = m.user_id 
GROUP BY u.id
ORDER BY  media_count ;


Ordina gli utenti per totale di likes ricevuti nei loro posts (25) 

SELECT u.username, COUNT(l.id) AS total_likes
FROM users u
JOIN posts p ON u.id = p.user_id
LEFT JOIN likes l ON p.id = l.post_id
GROUP BY u.id
ORDER BY total_likes DESC;


SELECT u.username , COUNT( l.user_id ) AS like_total
FROM users u 
LEFT JOIN likes l ON u.id = l.user_id 
GROUP BY u.id
ORDER BY  like_total ;
